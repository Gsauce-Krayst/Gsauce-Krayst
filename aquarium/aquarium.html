JavaScript

// Hunger system
function updateHunger() {
  document.querySelectorAll('.fish').forEach(fish => {
    let hunger = parseInt(fish.dataset.hunger);
    hunger = Math.max(0, hunger - 2);
    fish.dataset.hunger = hunger;
    
    const hungerBar = fish.querySelector('.hunger-level');
    hungerBar.style.width = `${hunger}%`;
    
    // Update hunger color
    hungerBar.parentElement.className = 'hunger-bar';
    if (hunger < 30) hungerBar.parentElement.classList.add('hunger-critical');
    else if (hunger < 60) hungerBar.parentElement.classList.add('hunger-low');
    
    // Fish dies if hunger reaches 0
    if (hunger === 0 && !fish.classList.contains('dead')) {
      fish.classList.add('dead');
      fish.style.animation = 'dead 2s forwards';
      setTimeout(() => fish.remove(), 2000);
    }
  });
  
  updateStatusPanel();
}

// Feed fish function
function feedFish() {
  document.querySelectorAll('.fish').forEach(fish => {
    if (!fish.classList.contains('dead')) {
      let hunger = parseInt(fish.dataset.hunger);
      hunger = Math.min(100, hunger + 30);
      fish.dataset.hunger = hunger;
      
      const hungerBar = fish.querySelector('.hunger-level');
      hungerBar.style.width = `${hunger}%`;
      
      // Animate fish when feeding
      fish.style.animation = 'wiggle 0.5s';
      setTimeout(() => fish.style.animation = '', 500);
    }
  });
  
  // Add food particles
  for (let i = 0; i < 5; i++) {
    createFoodParticle();
  }
}
